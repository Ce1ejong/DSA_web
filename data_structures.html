Explanation

In Python, the built-in list type acts as a dynamic array. The core operations are implemented using basic list methods wrapped in a class. Traversal simply involves iterating through all elements.

Python Code (`python_scripts/array.py`)
class ArrayOperations:
    def __init__(self, initial_list=None):
        self.array = initial_list if initial_list is not None else []

    def insert_element(self, index, value):
        self.array.insert(index, value)

    def delete_element(self, index):
        if 0 <= index < len(self.array):
            self.array.pop(index)

    def traverse(self):
        return self.array

# Example Usage:
arr_ops = ArrayOperations([10, 20, 30, 40])
print(f"Initial Array: {arr_ops.traverse()}\n")

arr_ops.insert_element(2, 50)  # Insert 50 at index 2
print(f"Array after inserting 50 at index 2: {arr_ops.traverse()}\n")

arr_ops.delete_element(3)      # Delete element at index 3 (was 30)
print(f"Array after deleting element at index 3: {arr_ops.traverse()}")

Explanation

A Stack is a LIFO (Last-In, First-Out) structure. Operations like Infix to Postfix conversion heavily rely on this structure.

Python Code (`python_scripts/stack.py`)
class Stack:
    def __init__(self): 
        self.items = []

    def is_empty(self): 
        return not self.items

    def push(self, item): 
        self.items.append(item) 

    def pop(self): 
        if not self.is_empty():
            return self.items.pop() 
        return "Stack is empty"
    
    def peek(self):
        if not self.is_empty():
            return self.items[-1]
        return "Stack is empty"

# Example Usage:
s = Stack()
s.push(10)
s.push(20)
s.push(30)
print(f"Items after pushes: {s.items}\n")
print(f"Peek (Top element): {s.peek()}\n")
print(f"Popped element: {s.pop()}\n")
print(f"Items after pop: {s.items}")

Explanation

A Queue is a FIFO (First-In, First-Out) structure, like a line of people waiting.

Python Code (`python_scripts/queue.py`)
from collections import deque

class Queue:
    def __init__(self):
        self.items = deque() 

    def enqueue(self, item):
        self.items.append(item) 

    def dequeue(self):
        if self.items:
            return self.items.popleft() 
        return "Queue is empty"
    
    def display(self):
        return list(self.items)

# Example Usage:
q = Queue()
q.enqueue(10)
q.enqueue(20)
q.enqueue(30)
print(f"Initial Queue: {q.display()}\n")

first_out = q.dequeue()
print(f"Dequeued item: {first_out}\n") 

print(f"Current Queue: {q.display()}\n")

q.enqueue(40)
print(f"Queue after Enqueue(40): {q.display()}")

Explanation

A Binary Tree is a non-linear data structure where each node has at most two children.

Python Code (`python_scripts/tree.py`)
class Node:
    def __init__(self, key):
        self.left = None
        self.right = None
        self.val = key

output_list = []
def print_output(val):
    global output_list
    output_list.append(str(val))

class BinaryTree:
    def __init__(self):
        global output_list
        output_list = [] 

    def inorder_traversal(self, root):
        if root:
            self.inorder_traversal(root.left)
            print_output(root.val)
            self.inorder_traversal(root.right)
    
    def preorder_traversal(self, root):
        if root:
            print_output(root.val)
            self.preorder_traversal(root.left)
            self.preorder_traversal(root.right)

# Example Usage:
root = Node(1)
root.left = Node(2)
root.right = Node(3)
root.left.left = Node(4)
root.left.right = Node(5)

tree = BinaryTree()
tree.inorder_traversal(root)
print(f"Inorder Traversal (L-R-R): {' '.join(output_list)}\n") 

tree_pre = BinaryTree()
tree_pre.preorder_traversal(root)
print(f"Preorder Traversal (R-L-R): {' '.join(output_list)}")
